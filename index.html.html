  <h1>Commandez votre boisson</h1>

  <select id="boisson">

  </select>
  <button onclick="envoyerCommande()">Commander</button>
  <div class="confirmation" id="message"></div>

  <script>
    const URL_SCRIPT = "COLLE_ICI_TON_URL";

    function envoyerCommande() {
      const nom = document.getElementById("nom").value;
      const boisson = document.getElementById("boisson").value;

      if (!nom || !boisson) {
        alert("Merci de remplir tous les champs !");
        return;
      }

      fetch(URL_SCRIPT, {
        method: "POST",
        mode: "no-cors",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ nom, boisson })
      });

      document.getElementById("message").textContent = "Commande envoyée 🎉";
      document.getElementById("nom").value = "";
      document.getElementById("boisson").value = "";
    }
  </script>
</body>
</html>
